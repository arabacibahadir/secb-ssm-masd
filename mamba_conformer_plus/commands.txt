# Mamba Conformer Plus commands

## Prepare windowed EEG data
python mamba_conformer_plus/mambaconformer.py --prepare --label-source state --focused-minutes 10 --unfocused-minutes 20 --mat-root EEGData --out-dir data/eeg_attention --window 512 --stride 256

## Train and evaluate
python mamba_conformer_plus/mambaconformer.py --out-dir data/eeg_attention --epochs 80 --batch-size 64 --use-amp --bidir-ssm --cosine --lr 5e-4 --time-mask-ratio 0.12 --channel-drop-prob 0.06

## Normalization ablation
python mamba_conformer_plus/mambaconformer.py --out-dir data/eeg_attention --epochs 80 --batch-size 64 --use-amp --bidir-ssm --cosine --no-normalize --lr 5e-4 --time-mask-ratio 0.10 --channel-drop-prob 0.05
